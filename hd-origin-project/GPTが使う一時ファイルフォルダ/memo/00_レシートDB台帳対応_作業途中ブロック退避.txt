const _receiptFs = require("fs");
const _receiptPath = require("path");
const _receiptConfig = require("../config");

function _receiptAbsolutePath(relativePath) {
  if (!relativePath) return "";

  return _receiptPath.join(
    _receiptConfig.receiptRoot,
    String(relativePath).replace(/[\\/]+/g, _receiptPath.sep)
  );
}

async function listImports(limit = 100, offset = 0) {
  const safeLimit = Math.min(Math.max(Number(limit) || 100, 1), 500);
  const safeOffset = Math.max(Number(offset) || 0, 0);

  const result = await pool.query(
    `
    SELECT
      ri.receipt_import_id AS id,
      'receipt-' || ri.receipt_import_id AS upload_id,
      NULL::text AS wix_item_id,
      NULL::text AS wix_image_url,
      rf.stored_file_name AS local_image_file_name,
      rf.relative_path,
      rf.storage_root_key,
      rf.file_status,
      rf.sha256 AS image_hash_sha256,
      rf.file_size AS image_size_bytes,
      ri.original_file_name,
      ri.created_at AS captured_at_jst,
      ri.created_at AS imported_at_jst,
      ri.import_source AS import_batch_id,
      ri.ocr_provider,
      ri.ocr_raw_text,
      CASE
        WHEN ri.ocr_raw_text IS NULL OR trim(ri.ocr_raw_text) = '' THEN 0
        ELSE array_length(regexp_split_to_array(trim(ri.ocr_raw_text), E'\\n+'), 1)
      END AS ocr_line_count,
      CASE
        WHEN ri.ocr_raw_text IS NULL OR trim(ri.ocr_raw_text) = '' THEN 0
        ELSE array_length(regexp_split_to_array(trim(ri.ocr_raw_text), E'\\s+'), 1)
      END AS ocr_word_count,
      ri.import_status AS status,
      ri.receipt_date,
      ri.vendor_name,
      ri.total_amount,
      ri.tax_amount,
      ri.invoice_number,
      ri.payment_method_id,
      ri.payment_method_name,
      ri.target_person_id,
      ri.target_person,
      ri.purpose_id,
      ri.purpose,
      ri.project_id,
      ri.project_name,
      ri.department_id,
      ri.department_name,
      ri.evidence_type,
      ri.evidence_memo,
      ri.summary,
      ri.created_at,
      ri.updated_at
    FROM expenses.receipt_imports ri
    LEFT JOIN LATERAL (
      SELECT *
      FROM expenses.receipt_files rf
      WHERE rf.receipt_import_id = ri.receipt_import_id
      ORDER BY rf.receipt_file_id
      LIMIT 1
    ) rf ON TRUE
    ORDER BY ri.created_at DESC NULLS LAST, ri.receipt_import_id DESC
    LIMIT $1 OFFSET $2
    `,
    [safeLimit, safeOffset]
  );

  return result.rows.map((row) => {
    const localImagePath = _receiptAbsolutePath(row.relative_path);

    return {
      ...row,
      local_image_path: localImagePath,
      file_missing: row.relative_path ? !_receiptFs.existsSync(localImagePath) : true,
    };
  });
}

async function getImportByImageHashSha256(hash) {
  const result = await pool.query(
    `
    SELECT
      ri.receipt_import_id AS id,
      'receipt-' || ri.receipt_import_id AS upload_id,
      rf.stored_file_name AS local_image_file_name,
      rf.relative_path,
      rf.sha256 AS image_hash_sha256,
      rf.file_size AS image_size_bytes,
      ri.original_file_name,
      ri.created_at AS imported_at_jst,
      ri.ocr_provider,
      ri.ocr_raw_text,
      ri.import_status AS status,
      ri.created_at,
      ri.updated_at
    FROM expenses.receipt_files rf
    JOIN expenses.receipt_imports ri
      ON ri.receipt_import_id = rf.receipt_import_id
    WHERE rf.sha256 = $1
    ORDER BY ri.receipt_import_id DESC
    LIMIT 1
    `,
    [hash]
  );

  const row = result.rows[0] || null;

  if (!row) return null;

  return {
    ...row,
    local_image_path: _receiptAbsolutePath(row.relative_path),
  };
}

async function createLocalImport(data) {
  const client = await pool.connect();

  try {
    await client.query("BEGIN");

    const relativePath = _receiptPath
      .relative(_receiptConfig.receiptRoot, data.localImagePath)
      .replace(/[\\]+/g, "/");

    const importResult = await client.query(
      `
      INSERT INTO expenses.receipt_imports (
        import_source,
        import_status,
        original_file_name,
        display_file_name,
        total_amount,
        evidence_type,
        ocr_provider,
        ocr_status,
        ocr_raw_text,
        summary,
        created_at,
        updated_at
      )
      VALUES (
        $1,
        'ready',
        $2,
        $3,
        0,
        'receipt',
        $4,
        'done',
        $5,
        $6,
        CURRENT_TIMESTAMP,
        CURRENT_TIMESTAMP
      )
      RETURNING *
      `,
      [
        data.importBatchId || "local",
        data.originalFileName || data.localImageFileName || "",
        data.localImageFileName || data.originalFileName || "",
        data.ocrProvider || "",
        data.ocrRawText || "",
        "ローカル取込: " + (data.localImageFileName || data.originalFileName || "")
      ]
    );

    const receiptImport = importResult.rows[0];

    await client.query(
      `
      INSERT INTO expenses.receipt_files (
        receipt_import_id,
        file_role,
        storage_root_key,
        original_file_name,
        stored_file_name,
        relative_path,
        mime_type,
        file_size,
        sha256,
        file_status,
        last_checked_at,
        created_at,
        updated_at
      )
      VALUES (
        $1,
        'receipt_image',
        'HD_ORIGIN_RECEIPT_ROOT',
        $2,
        $3,
        $4,
        $5,
        $6,
        $7,
        'ready',
        CURRENT_TIMESTAMP,
        CURRENT_TIMESTAMP,
        CURRENT_TIMESTAMP
      )
      `,
      [
        receiptImport.receipt_import_id,
        data.originalFileName || "",
        data.localImageFileName || "",
        relativePath,
        data.mimeType || "image/jpeg",
        data.imageSizeBytes || null,
        data.imageHashSha256 || null
      ]
    );

    await client.query("COMMIT");

    return {
      ...receiptImport,
      id: receiptImport.receipt_import_id,
      local_image_file_name: data.localImageFileName || "",
      local_image_path: data.localImagePath || "",
      image_hash_sha256: data.imageHashSha256 || "",
      image_size_bytes: data.imageSizeBytes || null,
      original_file_name: data.originalFileName || "",
      imported_at_jst: receiptImport.created_at,
      status: receiptImport.import_status
    };
  } catch (error) {
    await client.query("ROLLBACK").catch(() => {});
    throw error;
  } finally {
    client.release();
  }
}

async function listImportsForOcrDuplicateCheck(limit = 300) {
  const safeLimit = Math.min(Math.max(Number(limit) || 300, 1), 1000);

  const result = await pool.query(
    `
    SELECT
      ri.receipt_import_id AS id,
      'receipt-' || ri.receipt_import_id AS upload_id,
      rf.stored_file_name AS local_image_file_name,
      rf.relative_path,
      rf.sha256 AS image_hash_sha256,
      ri.original_file_name,
      ri.created_at AS imported_at_jst,
      ri.ocr_provider,
      ri.ocr_raw_text,
      ri.import_status AS status,
      ri.created_at,
      ri.updated_at
    FROM expenses.receipt_imports ri
    LEFT JOIN LATERAL (
      SELECT *
      FROM expenses.receipt_files rf
      WHERE rf.receipt_import_id = ri.receipt_import_id
      ORDER BY rf.receipt_file_id
      LIMIT 1
    ) rf ON TRUE
    WHERE ri.ocr_raw_text IS NOT NULL
      AND length(trim(ri.ocr_raw_text)) > 0
    ORDER BY ri.receipt_import_id DESC
    LIMIT $1
    `,
    [safeLimit]
  );

  return result.rows.map((row) => ({
    ...row,
    local_image_path: _receiptAbsolutePath(row.relative_path),
  }));
}

async function deleteImportById(id) {
  const client = await pool.connect();

  try {
    await client.query("BEGIN");

    const found = await client.query(
      `
      SELECT *
      FROM expenses.receipt_imports
      WHERE receipt_import_id = $1
      LIMIT 1
      `,
      [id]
    );

    const item = found.rows[0] || null;

    if (!item) {
      await client.query("ROLLBACK");
      return null;
    }

    await client.query(
      `
      DELETE FROM expenses.receipt_imports
      WHERE receipt_import_id = $1
      `,
      [id]
    );

    await client.query("COMMIT");

    return {
      ...item,
      id: item.receipt_import_id
    };
  } catch (error) {
    await client.query("ROLLBACK").catch(() => {});
    throw error;
  } finally {
    client.release();
  }
}
